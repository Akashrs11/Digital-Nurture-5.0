-- Scenario 1: Update customer last modified date automatically

CREATE OR REPLACE TRIGGER UpdateCustomerLastModified
BEFORE UPDATE ON Customers
FOR EACH ROW

BEGIN

    :NEW.LastModified := SYSDATE;

END;
/
-- Create AuditLog table for transaction history

CREATE TABLE AuditLog
(
    LogID NUMBER GENERATED ALWAYS AS IDENTITY,
    TransactionID NUMBER,
    ActionDate DATE,
    ActionType VARCHAR2(50)
);


CREATE OR REPLACE TRIGGER LogTransaction
AFTER INSERT ON Transactions
FOR EACH ROW

BEGIN

    INSERT INTO AuditLog
    (
        TransactionID,
        ActionDate,
        ActionType
    )
    VALUES
    (
        :NEW.TransactionID,
        SYSDATE,
        'Transaction Added'
    );

END;
/
-- Scenario 3: Check deposit and withdrawal rules

CREATE OR REPLACE TRIGGER CheckTransactionRules
BEFORE INSERT ON Transactions
FOR EACH ROW

DECLARE

    current_balance NUMBER;

BEGIN


    SELECT Balance
    INTO current_balance
    FROM Accounts
    WHERE AccountID = :NEW.AccountID;



    IF :NEW.TransactionType = 'Withdrawal'
       AND :NEW.Amount > current_balance THEN

        RAISE_APPLICATION_ERROR
        (
        -20001,
        'Withdrawal exceeds balance'
        );

    END IF;



    IF :NEW.TransactionType = 'Deposit'
       AND :NEW.Amount <= 0 THEN

        RAISE_APPLICATION_ERROR
        (
        -20002,
        'Deposit amount must be positive'
        );

    END IF;


END;
/
